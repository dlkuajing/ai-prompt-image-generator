优化后的下载函数

新的下载函数使用 Blob 和 createObjectURL 方法，确保图片能够正确下载而不是在新标签页打开。

async function downloadImage(url, filename) { try { // 获取图片数据 const response = await fetch(url); const blob = await response.blob(); // 创建blob URL const blobUrl = window.URL.createObjectURL(blob); // 创建下载链接 const a = document.createElement('a'); a.href = blobUrl; a.download = filename || \`image_\${Date.now()}.png\`; document.body.appendChild(a); a.click(); // 清理 document.body.removeChild(a); window.URL.revokeObjectURL(blobUrl); showMessage('图片下载成功！'); } catch (error) { console.error('下载失败:', error); // 备用方案... } }
